// ==================== COMPONENTE EVALUADOR SPICE ====================

const EvaluadorSPICE = () => {
  const [evaluacion, setEvaluacion] = useState({
    desarrollo: 0,
    gestionProyectos: 0,
    gestionCalidad: 0,
    gestionConfiguracion: 0,
    verificacion: 0
  });

  const [resultado, setResultado] = useState(null);

  const nivelesCapacidad = [
    { nivel: 0, nombre: 'Incompleto', color: '#ef4444', descripcion: 'Proceso no implementado o falla en lograr su propósito' },
    { nivel: 1, nombre: 'Realizado', color: '#f97316', descripcion: 'El proceso logra su propósito' },
    { nivel: 2, nombre: 'Gestionado', color: '#eab308', descripcion: 'Planificado, monitoreado y ajustado' },
    { nivel: 3, nombre: 'Establecido', color: '#84cc16', descripcion: 'Implementado usando proceso definido' },
    { nivel: 4, nombre: 'Predecible', color: '#22c55e', descripcion: 'Opera dentro de límites definidos' },
    { nivel: 5, nombre: 'Optimizado', color: '#10b981', descripcion: 'Mejorado continuamente para objetivos' }
  ];

  const procesosEvaluados = [
    { key: 'desarrollo', nombre: 'Desarrollo de Software', icono: Code },
    { key: 'gestionProyectos', nombre: 'Gestión de Proyectos', icono: Target },
    { key: 'gestionCalidad', nombre: 'Aseguramiento de Calidad', icono: Award },
    { key: 'gestionConfiguracion', nombre: 'Gestión de Configuración', icono: Shield },
    { key: 'verificacion', nombre: 'Verificación y Validación', icono: CheckCircle }
  ];

  const handleEvaluacion = (proceso, nivel) => {
    setEvaluacion(prev => ({ ...prev, [proceso]: nivel }));
  };

  const calcularResultados = () => {
    const valores = Object.values(evaluacion);
    const promedio = valores.reduce((a, b) => a + b, 0) / valores.length;

    const datosGrafico = procesosEvaluados.map(p => ({
      proceso: p.nombre,
      nivel: evaluacion[p.key],
      fill: nivelesCapacidad[evaluacion[p.key]].color
    }));

    const recomendaciones = [];

    procesosEvaluados.forEach(p => {
      const nivel = evaluacion[p.key];
      if (nivel < 2) {
        recomendaciones.push({
          tipo: 'error',
          proceso: p.nombre,
          mensaje: `Nivel ${nivel} (${nivelesCapacidad[nivel].nombre}). Prioridad ALTA: Establecer planificación y seguimiento básico.`
        });
      } else if (nivel < 3) {
        recomendaciones.push({
          tipo: 'warning',
          proceso: p.nombre,
          mensaje: `Nivel ${nivel} (${nivelesCapacidad[nivel].nombre}). Documentar y estandarizar el proceso.`
        });
      } else if (nivel >= 4) {
        recomendaciones.push({
          tipo: 'success',
          proceso: p.nombre,
          mensaje: `Nivel ${nivel} (${nivelesCapacidad[nivel].nombre}). ¡Excelente! Proceso maduro y bajo control.`
        });
      }
    });

    let nivelOrganizacional = '';
    if (promedio < 1.5) nivelOrganizacional = 'Inicial - Requiere Mejora Urgente';
    else if (promedio < 2.5) nivelOrganizacional = 'Gestionado - En Desarrollo';
    else if (promedio < 3.5) nivelOrganizacional = 'Establecido - Buena Base';
    else if (promedio < 4.5) nivelOrganizacional = 'Predecible - Alto Rendimiento';
    else nivelOrganizacional = 'Optimizado - Excelencia';

    setResultado({
      promedio: promedio.toFixed(2),
      nivelOrganizacional,
      datosGrafico,
      recomendaciones
    });
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-orange-50 to-purple-100 p-8">
      <div className="max-w-6xl mx-auto">

        <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
          <div className="flex items-center space-x-4 mb-4">
            <div className="p-3 bg-gradient-to-r from-orange-500 to-purple-600 rounded-lg">
              <BarChart3 className="w-8 h-8 text-white" />
            </div>
            <div>
              <h1 className="text-4xl font-bold text-gray-800">Evaluador SPICE</h1>
              <p className="text-gray-600">ISO/IEC 15504 - Evaluación de Capacidad de Procesos</p>
            </div>
          </div>

          <div className="bg-orange-50 border-l-4 border-orange-500 p-4 rounded">
            <div className="flex items-start space-x-3">
              <Info className="w-5 h-5 text-orange-600 flex-shrink-0 mt-0.5" />
              <div className="text-sm text-orange-800">
                <p className="font-semibold mb-1">¿Qué es SPICE?</p>
                <p>SPICE (Software Process Improvement and Capability Determination) evalúa la <strong>capacidad de tus procesos</strong> en 6 niveles, desde "Incompleto" hasta "Optimizado". A diferencia de CMMI que evalúa toda la organización, SPICE permite evaluar procesos individuales.</p>
              </div>
            </div>
          </div>
        </div>

        <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
          <h2 className="text-2xl font-bold text-gray-800 mb-6 flex items-center">
            <Target className="w-6 h-6 mr-2 text-purple-600" />
            Evalúa la Capacidad de tus Procesos
          </h2>

          <div className="space-y-6">
            {procesosEvaluados.map(proceso => {
              const Icon = proceso.icono;
              return (
                <div key={proceso.key} className="border rounded-lg p-6 hover:shadow-md transition-shadow">
                  <div className="flex items-center mb-4">
                    <Icon className="w-6 h-6 text-purple-600 mr-3" />
                    <h3 className="text-lg font-semibold text-gray-800">{proceso.nombre}</h3>
                  </div>

                  <div className="flex flex-wrap gap-2">
                    {nivelesCapacidad.map((nivel) => (
                      <button
                        key={nivel.nivel}
                        onClick={() => handleEvaluacion(proceso.key, nivel.nivel)}
                        className={`px-4 py-2 rounded-lg font-medium transition-all ${evaluacion[proceso.key] === nivel.nivel
                            ? 'ring-2 ring-offset-2 ring-purple-500 scale-105'
                            : 'hover:scale-105'
                          }`}
                        style={{
                          backgroundColor: evaluacion[proceso.key] === nivel.nivel ? nivel.color : '#f3f4f6',
                          color: evaluacion[proceso.key] === nivel.nivel ? 'white' : '#374151'
                        }}
                        title={nivel.descripcion}
                      >
                        {nivel.nivel} - {nivel.nombre}
                      </button>
                    ))}
                  </div>
                  <p className="text-sm text-gray-500 mt-2">
                    {nivelesCapacidad[evaluacion[proceso.key]].descripcion}
                  </p>
                </div>
              );
            })}
          </div>

          <button
            onClick={calcularResultados}
            className="mt-6 w-full bg-gradient-to-r from-orange-600 to-purple-600 text-white font-bold py-4 rounded-lg hover:from-orange-700 hover:to-purple-700 transition-all transform hover:scale-[1.02] shadow-lg flex items-center justify-center space-x-2"
          >
            <BarChart3 className="w-5 h-5" />
            <span>Generar Informe de Capacidad</span>
          </button>
        </div>

        {resultado && (
          <>
            <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
              <div className="text-center">
                <div className="inline-block p-4 bg-gradient-to-r from-orange-500 to-purple-500 rounded-full mb-4">
                  <Award className="w-12 h-12 text-white" />
                </div>
                <h2 className="text-3xl font-bold text-gray-800 mb-2">Nivel Organizacional</h2>
                <div className="text-6xl font-bold bg-gradient-to-r from-orange-600 to-purple-600 bg-clip-text text-transparent mb-4">
                  {resultado.promedio}/5.0
                </div>
                <span className="inline-block px-6 py-2 rounded-full font-bold text-lg bg-purple-50 text-purple-800 border-2 border-purple-300">
                  {resultado.nivelOrganizacional}
                </span>
              </div>
            </div>

            <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
              <h3 className="text-2xl font-bold text-gray-800 mb-6 text-center">
                Niveles de Capacidad por Proceso
              </h3>
              <div className="h-96">
                <ResponsiveContainer width="100%" height="100%">
                  <BarChart data={resultado.datosGrafico}>
                    <CartesianGrid strokeDasharray="3 3" />
                    <XAxis dataKey="proceso" angle={-15} textAnchor="end" height={100} />
                    <YAxis domain={[0, 5]} ticks={[0, 1, 2, 3, 4, 5]} />
                    <RechartsTooltip />
                    <Bar dataKey="nivel" radius={[8, 8, 0, 0]}>
                      {resultado.datosGrafico.map((entry, index) => (
                        <Cell key={`cell-${index}`} fill={entry.fill} />
                      ))}
                    </Bar>
                  </BarChart>
                </ResponsiveContainer>
              </div>
            </div>

            <div className="bg-white rounded-xl shadow-lg p-8">
              <h3 className="text-2xl font-bold text-gray-800 mb-6 flex items-center">
                <Lightbulb className="w-6 h-6 mr-2 text-yellow-500" />
                Plan de Mejora Recomendado
              </h3>
              <div className="space-y-4">
                {resultado.recomendaciones.map((rec, idx) => {
                  const Icon = rec.tipo === 'success' ? CheckCircle : rec.tipo === 'error' ? AlertCircle : Info;
                  const colorClass = rec.tipo === 'success' ? 'bg-green-50 border-green-500 text-green-800' :
                    rec.tipo === 'error' ? 'bg-red-50 border-red-500 text-red-800' :
                      'bg-yellow-50 border-yellow-500 text-yellow-800';

                  return (
                    <div key={idx} className={`border-l-4 p-4 rounded ${colorClass}`}>
                      <div className="flex items-start space-x-3">
                        <Icon className="w-5 h-5 flex-shrink-0 mt-0.5" />
                        <div>
                          <p className="font-bold">{rec.proceso}</p>
                          <p className="text-sm mt-1">{rec.mensaje}</p>
                        </div>
                      </div>
                    </div>
                  );
                })}
              </div>
            </div>
          </>
        )}
      </div>
    </div>
  );
};

// ==================== COMPONENTE CALCULADORA ====================

const CalculadoraMetricas = () => {
  const [metricas, setMetricas] = useState({
    mtbf: '',
    mttr: '',
    tiempoRespuesta: '',
    complejidad: '',
    coberturaPruebas: ''
  });

  const [resultado, setResultado] = useState(null);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setMetricas(prev => ({
      ...prev,
      [name]: value
    }));
  };

  const calcularScore = () => {
    const mtbf = parseFloat(metricas.mtbf) || 0;
    const mttr = parseFloat(metricas.mttr) || 0;
    const tiempoRespuesta = parseFloat(metricas.tiempoRespuesta) || 0;
    const complejidad = parseFloat(metricas.complejidad) || 0;
    const cobertura = parseFloat(metricas.coberturaPruebas) || 0;

    const scoreFiabilidad = Math.min(100, ((mtbf / 720) * 50) + ((1 - (mttr / 60)) * 50));
    const scoreEficiencia = Math.max(0, 100 - (tiempoRespuesta / 30) * 100);
    const scoreMantenibilidad = Math.max(0, 100 - (complejidad * 5));
    const scoreFuncionalidad = cobertura;
    const scoreSeguridad = (scoreFiabilidad + scoreMantenibilidad) / 2;

    const scoreGeneral = (
      scoreFiabilidad * 0.25 +
      scoreEficiencia * 0.20 +
      scoreMantenibilidad * 0.20 +
      scoreFuncionalidad * 0.20 +
      scoreSeguridad * 0.15
    );

    const datosRadar = [
      { caracteristica: 'Fiabilidad', valor: Math.round(scoreFiabilidad), benchmark: 85 },
      { caracteristica: 'Eficiencia', valor: Math.round(scoreEficiencia), benchmark: 80 },
      { caracteristica: 'Mantenibilidad', valor: Math.round(scoreMantenibilidad), benchmark: 75 },
      { caracteristica: 'Funcionalidad', valor: Math.round(scoreFuncionalidad), benchmark: 90 },
      { caracteristica: 'Seguridad', valor: Math.round(scoreSeguridad), benchmark: 85 }
    ];

    const recomendaciones = [];

    if (scoreFiabilidad < 70) {
      recomendaciones.push({
        tipo: 'error',
        area: 'Fiabilidad',
        mensaje: 'MTBF bajo o MTTR alto. Implementa pruebas de regresión y mejora la gestión de errores.'
      });
    }

    if (scoreEficiencia < 70) {
      recomendaciones.push({
        tipo: 'warning',
        area: 'Eficiencia',
        mensaje: 'Tiempo de respuesta elevado. Optimiza consultas a BD y considera implementar caché.'
      });
    }

    if (scoreMantenibilidad < 70) {
      recomendaciones.push({
        tipo: 'error',
        area: 'Mantenibilidad',
        mensaje: 'Complejidad ciclomática alta. Refactoriza funciones complejas en módulos más pequeños.'
      });
    }

    if (scoreFuncionalidad < 80) {
      recomendaciones.push({
        tipo: 'warning',
        area: 'Funcionalidad',
        mensaje: 'Cobertura de pruebas insuficiente. Objetivo: >80% para código crítico.'
      });
    }

    if (recomendaciones.length === 0) {
      recomendaciones.push({
        tipo: 'success',
        area: 'General',
        mensaje: '¡Excelente! Tu software cumple con los estándares de calidad ISO/IEC 25010.'
      });
    }

    setResultado({
      scoreGeneral: Math.round(scoreGeneral),
      datosRadar,
      recomendaciones,
      nivel: scoreGeneral >= 85 ? 'Excelente' : scoreGeneral >= 70 ? 'Bueno' : scoreGeneral >= 50 ? 'Aceptable' : 'Necesita Mejoras'
    });
  };

  const getNivelColor = (nivel) => {
    switch (nivel) {
      case 'Excelente': return 'text-green-600 bg-green-50';
      case 'Bueno': return 'text-blue-600 bg-blue-50';
      case 'Aceptable': return 'text-yellow-600 bg-yellow-50';
      default: return 'text-red-600 bg-red-50';
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 p-8">
      <div className="max-w-6xl mx-auto">

        <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
          <div className="flex items-center space-x-4 mb-4">
            <div className="p-3 bg-gradient-to-r from-blue-500 to-indigo-600 rounded-lg">
              <Calculator className="w-8 h-8 text-white" />
            </div>
            <div>
              <h1 className="text-4xl font-bold text-gray-800">Calculadora de Métricas</h1>
              <p className="text-gray-600">ISO/IEC 25010 - Modelo de Calidad del Producto</p>
            </div>
          </div>

          <div className="bg-blue-50 border-l-4 border-blue-500 p-4 rounded">
            <div className="flex items-start space-x-3">
              <Info className="w-5 h-5 text-blue-600 flex-shrink-0 mt-0.5" />
              <div className="text-sm text-blue-800">
                <p className="font-semibold mb-1">Sobre esta calculadora:</p>
                <p>Evalúa la calidad de tu software según 5 características clave del estándar ISO/IEC 25010.</p>
              </div>
            </div>
          </div>
        </div>

        <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
          <h2 className="text-2xl font-bold text-gray-800 mb-6 flex items-center">
            <TrendingUp className="w-6 h-6 mr-2 text-indigo-600" />
            Ingresa las Métricas de tu Proyecto
          </h2>

          <div className="grid md:grid-cols-2 gap-6">
            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">
                MTBF - Tiempo Medio Entre Fallos (horas)
              </label>
              <input
                type="number"
                name="mtbf"
                value={metricas.mtbf}
                onChange={handleChange}
                placeholder="ej. 720 (30 días)"
                className="w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition"
              />
              <p className="text-xs text-gray-500 mt-1">Recomendado: &gt;720 horas</p>
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">
                MTTR - Tiempo Medio de Recuperación (minutos)
              </label>
              <input
                type="number"
                name="mttr"
                value={metricas.mttr}
                onChange={handleChange}
                placeholder="ej. 15"
                className="w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition"
              />
              <p className="text-xs text-gray-500 mt-1">Objetivo: &lt;30 minutos</p>
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">
                Tiempo de Respuesta p95 (milisegundos)
              </label>
              <input
                type="number"
                name="tiempoRespuesta"
                value={metricas.tiempoRespuesta}
                onChange={handleChange}
                placeholder="ej. 200"
                className="w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition"
              />
              <p className="text-xs text-gray-500 mt-1">Objetivo: &lt;300 ms</p>
            </div>

            <div>
              <label className="block text-sm font-semibold text-gray-700 mb-2">
                Complejidad Ciclomática Promedio
              </label>
              <input
                type="number"
                name="complejidad"
                value={metricas.complejidad}
                onChange={handleChange}
                placeholder="ej. 8"
                className="w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition"
              />
              <p className="text-xs text-gray-500 mt-1">Recomendado: &lt;10</p>
            </div>

            <div className="md:col-span-2">
              <label className="block text-sm font-semibold text-gray-700 mb-2">
                Cobertura de Pruebas (%)
              </label>
              <input
                type="number"
                name="coberturaPruebas"
                value={metricas.coberturaPruebas}
                onChange={handleChange}
                placeholder="ej. 85"
                min="0"
                max="100"
                className="w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent transition"
              />
              <p className="text-xs text-gray-500 mt-1">Objetivo: &gt;80%</p>
            </div>
          </div>

          <button
            onClick={calcularScore}
            className="mt-6 w-full bg-gradient-to-r from-indigo-600 to-blue-600 text-white font-bold py-4 rounded-lg hover:from-indigo-700 hover:to-blue-700 transition-all transform hover:scale-[1.02] shadow-lg flex items-center justify-center space-x-2"
          >
            <Calculator className="w-5 h-5" />
            <span>Calcular Score de Calidad</span>
          </button>
        </div>

        {resultado && (
          <>
            <div className="bg-white rounded-xl shadow-lg p-8 mb-8">
              <div className="text-center">
                <div className="inline-block p-4 bg-gradient-to-r from-indigo-500 to-blue-500 rounded-full mb-4">
                  <Award className="w-12 h-12 text-white" />
